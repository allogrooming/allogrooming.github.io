to_do 테이블 생성 및 테스트 데이터 추가 
```
create table to_do(
	to_do_code int not null,
    member_code int not null,
    to_do_content varchar(20) not null,
    set_up_date date not null,
    to_do_state int,
    to_do_color varchar(20) not null,
    done_date date,
    plan_date date,
    primary key (to_do_code));
```
